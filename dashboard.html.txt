        function generatePrediction() {
            const currentPrice = parseFloat(document.getElementById('currentPrice').value);
            const days = parseInt(document.getElementById('daysToExpiry').value);
            const volatility = parseFloat(document.getElementById('volatility').value);
            
            // Simple prediction simulation
            const randomFactor = (Math.random() - 0.5) * 0.02;
            const volatilityFactor = (volatility / 100) * Math.sqrt(days / 252);
            const predictedChange = randomFactor + volatilityFactor;
            const predictedPrice = currentPrice * (1 + predictedChange);
            
            // Update strikes based on prediction
            updateIronCondorStrikes(predictedPrice);
            
            // Show prediction result
            const resultDiv = document.getElementById('predictionResult');
            const predictionText = document.getElementById('predictionText');
            
            predictionText.innerHTML = `
                <strong>Predicted SPX Price in ${days} days: $${predictedPrice.toFixed(2)}</strong><br>
                Expected change: ${(predictedChange * 100).toFixed(2)}%<br>
                Confidence: ${(65 + Math.random() * 20).toFixed(1)}%
            `;
            
            resultDiv.style.display = 'block';
        }

        function updateIronCondorStrikes(predictedPrice) {
            const spread = 50;
            const buffer = 25;
            
            document.getElementById('putShort').textContent = Math.round(predictedPrice - buffer);
            document.getElementById('putLong').textContent = Math.round(predictedPrice - buffer - spread);
            document.getElementById('callShort').textContent = Math.round(predictedPrice + buffer);
            document.getElementById('callLong').textContent = Math.round(predictedPrice + buffer + spread);
        }

        function logout() {
            window.location.href = 'index.html';
        }

        // Initialize with current data
        document.addEventListener('DOMContentLoaded', function() {
            generatePrediction();
        });
